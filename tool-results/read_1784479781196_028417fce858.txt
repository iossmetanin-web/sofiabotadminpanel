   151→async def get_user(user_id: int) -> Optional[dict]:
   152→    """Возвращает пользователя как dict или None."""
   153→    conn = await get_conn()

   155→        row = await conn.fetchrow("SELECT * FROM users WHERE user_id = $1", user_id)

   158→        await conn.close()

   198→async def update_user_profile(user_id: int, **fields):
   199→    """Универсальное обновление профиля."""
   200→    if not fields:
   201→        return
   202→    conn = await get_conn()
   203→    try:
   204→        sets = []
   205→        vals = []
   206→        for k, v in fields.items():
   207→            sets.append(f"{k} = ${len(vals) + 1}")
   208→            vals.append(v)
   209→        vals.append(user_id)
   210→        query = f"UPDATE users SET {', '.join(sets)}, updated_at = NOW() WHERE user_id = ${len(vals)}"
   211→        await conn.execute(query, *vals)
   212→    finally:
   213→        await conn.close()

   274→async def increment_daily_message_count(user_id: int) -> int:
   275→    """Увеличивает дневной счётчик сообщений.
   276→
   277→    Если последний раз считали в другой день — сбрасываем в 1.
   278→    Возвращает текущее количество сообщений за сегодня.
   279→    """
   280→    conn = await get_conn()
   281→    try:
   282→        new_count = await conn.fetchval(
   283→            """UPDATE users
   284→               SET daily_message_count = CASE
   285→                       WHEN daily_message_date = CURRENT_DATE THEN daily_message_count + 1
   286→                       ELSE 1
   287→                   END,
   288→                   daily_message_date = CURRENT_DATE,
   289→                   updated_at = NOW()
   290→               WHERE user_id = $1
   291→               RETURNING daily_message_count""",
   292→            user_id,
   293→        )
   294→        return new_count or 0
   295→    finally:
   296→        await conn.close()

   299→async def get_daily_message_count(user_id: int) -> int:
   300→    """Возвращает количество сообщений за сегодня (0 если в другой день)."""
   301→    conn = await get_conn()
   302→    try:
   303→        row = await conn.fetchrow(
   304→            """SELECT daily_message_count, daily_message_date FROM users WHERE user_id = $1""",
   305→            user_id,
   306→        )
   307→        if not row:
   308→            return 0
   309→        if row["daily_message_date"] is None:
   310→            return 0
   311→        # Если дата не сегодня — возвращаем 0
   312→        from datetime import date as _date
   313→        if row["daily_message_date"] != _date.today():
   314→            return 0
   315→        return row["daily_message_count"] or 0
   316→    finally:
   317→        await conn.close()

   356→async def spend_crystals(user_id: int, amount: int, description: str = "") -> bool:
   357→    conn = await get_conn()
   358→    try:
   359→        async with conn.transaction():
   360→            current = await conn.fetchval(
   361→                "SELECT crystals FROM users WHERE user_id = $1 FOR UPDATE", user_id
   362→            )
   363→            if current is None or current < amount:
   364→                return False
   365→            await conn.execute(
   366→                "UPDATE users SET crystals = crystals - $1, updated_at = NOW() WHERE user_id = $2",
   367→                amount, user_id,
   368→            )
   369→            await conn.execute(
   370→                "INSERT INTO transactions (user_id, type, amount, description) VALUES ($1, 'spend', $2, $3)",
   371→                user_id, amount, description,
   372→            )
   373→            logger.info(f"User {user_id} spent {amount} crystals: {description}")
   374→            return True
   375→    finally:
   376→        await conn.close()

   589→# ─── Rate Limiting (через БД, не в памяти) ───
   590→
   591→async def check_rate_limit(user_id: int, min_interval_seconds: float = 2.0) -> bool:
   592→    conn = await get_conn()
   593→    try:
   594→        row = await conn.fetchrow(
   595→            "SELECT last_message_at FROM rate_limits WHERE user_id = $1", user_id
   596→        )
   597→        if row:
   598→            last = row["last_message_at"]
   599→            if last:
   600→                from datetime import datetime as _dt, timezone as _tz
   601→                if hasattr(last, "tzinfo") and last.tzinfo is None:
   602→                    last = last.replace(tzinfo=_tz.utc)
   603→                diff = (_dt.now(_tz.utc) - last).total_seconds()
   604→                if diff < min_interval_seconds:
   605→                    return False
   606→        return True
   607→    finally:
   608→        await conn.close()

   749→# ─── Admin analytics (расширенная) ───
   750→
   751→async def get_admin_analytics() -> dict:
   752→    """Сводная статистика для /stats. Оптимизировано — один запрос вместо 12."""
   753→    conn = await get_conn()
   754→    try:
   755→        # Единый запрос с несколькими агрегатами — вместо 12 отдельных коннектов
   756→        row = await conn.fetchrow("""
   757→            SELECT
   758→                (SELECT COUNT(*) FROM users) AS total_users,
   759→                (SELECT COUNT(*) FROM users WHERE last_seen_at > NOW() - INTERVAL '24 hours') AS active_24h,
   760→                (SELECT COUNT(*) FROM users WHERE last_seen_at > NOW() - INTERVAL '7 days') AS active_7d,
   761→                (SELECT COUNT(*) FROM users WHERE onboarding_completed = TRUE) AS onboarding_done,
   762→                (SELECT COALESCE(SUM(crystals), 0) FROM users) AS total_crystals,
   763→                (SELECT COUNT(*) FROM conversations) AS total_messages,
   764→                (SELECT COUNT(*) FROM transactions WHERE type = 'spend') AS paid_transactions,
   765→                (SELECT COUNT(DISTINCT user_id) FROM transactions WHERE type = 'spend') AS paying_users,
   766→                (SELECT COUNT(*) FROM users WHERE is_blocked = TRUE) AS blocked_users,
   767→                (SELECT COUNT(*) FROM users WHERE referred_by IS NOT NULL) AS referral_users,
   768→                (SELECT COUNT(*) FROM users WHERE daily_horoscope_opt_in = TRUE) AS daily_subscribers
   769→        """)
   770→
   771→        total_users = row["total_users"] or 0
   772→        paying_users = row["paying_users"] or 0
   773→        conversion = round((paying_users / total_users * 100), 1) if total_users else 0.0
   774→
   775→        # Топ-5 активных (второй запрос)
   776→        top_active = await conn.fetch("""
   777→            SELECT name, first_name, username, message_count, crystals
   778→            FROM users WHERE is_blocked = FALSE
   779→            ORDER BY message_count DESC LIMIT 5
   780→        """)
   781→        top_active = [dict(r) for r in top_active]
   782→
   783→        return {
   784→            "total_users": total_users,
   785→            "active_24h": row["active_24h"] or 0,
   786→            "active_7d": row["active_7d"] or 0,
   787→            "onboarding_done": row["onboarding_done"] or 0,
   788→            "total_crystals": row["total_crystals"] or 0,
   789→            "total_messages": row["total_messages"] or 0,
   790→            "paid_transactions": row["paid_transactions"] or 0,
   791→            "paying_users": paying_users,
   792→            "conversion_pct": conversion,
   793→            "blocked_users": row["blocked_users"] or 0,
   794→            "referral_users": row["referral_users"] or 0,
   795→            "daily_subscribers": row["daily_subscribers"] or 0,
   796→            "top_active": top_active,
   797→        }
   798→    finally:
   799→        await conn.close()
