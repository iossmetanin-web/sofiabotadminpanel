   186→# ──────────────────────── Gemini Provider ────────────────────────
   187→
   188→async def _gemini_single_request(
   189→    model: str, api_key: str, contents: list[dict], timeout: float = 8.0,
   190→    max_tokens: int = 700
   191→) -> tuple[str, int]:
   192→    """Один запрос к Gemini API. Возвращает (text, status_code)."""
   193→    url = f"https://generativelanguage.googleapis.com/v1beta/models/{model}:generateContent"
   194→    payload = {
   195→        "contents": contents,
   196→        "generationConfig": {"temperature": 0.85, "maxOutputTokens": max_tokens, "topP": 0.92},
   197→    }
   198→    try:
   199→        async with httpx.AsyncClient(timeout=timeout) as client:
   200→            resp = await client.post(f"{url}?key={api_key}", headers={"Content-Type": "application/json"}, json=payload)
   201→            if resp.status_code == 404:
   202→                logger.error(f"Gemini model {model} not found (404)")
   203→                return ("", 404)
   204→            if resp.status_code == 429:
   205→                logger.warning(f"Gemini quota exhausted for {model} (429)")
   206→                return ("", 429)
   207→            resp.raise_for_status()
   208→            data = resp.json()
   209→            if data.get("candidates") and data["candidates"][0].get("content"):
   210→                parts = data["candidates"][0]["content"].get("parts", [])
   211→                if parts and parts[0].get("text"):
   212→                    return (parts[0]["text"].strip(), 200)
   213→            logger.warning(f"Gemini no text: {json.dumps(data, ensure_ascii=False)[:200]}")
   214→            return ("Извини, мне нужно подумать... Попробуй ещё раз.", 200)
   215→    except httpx.TimeoutException:
   216→        return ("", -1)
   217→    except Exception as e:
   218→        logger.error(f"Gemini error ({model}): {e}")
   219→        return ("", -2)
   220→
   221→
   222→async def _gemini_call(contents: list[dict], timeout: float = 8.0, max_tokens: int = 700) -> str:
   223→    """Gemini с fallback на другие модели."""
   224→    api_key = _get_api_key()
   225→    if not api_key:
   226→        return "У меня нет доступа к внутреннему голосу... Попробуй позже."
   227→
   228→    model = os.getenv("GEMINI_MODEL", GEMINI_MODELS[0])
   229→    text, status = await _gemini_single_request(model, api_key, contents, timeout, max_tokens)
   230→    if status == 200:
   231→        return text
   232→
   233→    if status in (404, 429):
   234→        for fallback in GEMINI_MODELS:
   235→            if fallback == model:
   236→                continue
   237→            logger.info(f"Gemini fallback: {fallback}")
   238→            text, status = await _gemini_single_request(fallback, api_key, contents, timeout, max_tokens)
   239→            if status == 200:
   240→                return text
   241→            if status == 429:
   242→                break
   243→        logger.error("All Gemini models exhausted")
   244→        return "Силы пока на исходе, милый человек. Подожди немного — и я снова буду готова говорить."
   245→
   246→    if status == -1:
   247→        return "Туман сегодня густой... Не успеваю разглядеть ответ. Попробуй ещё раз."
   248→    return "Что-то сегодня туман в голове... Попробуй сказать ещё раз."

   251→# ──────────────────────── OpenAI-Compatible Provider (Groq / OpenRouter) ────
   252→
   253→async def _openai_compatible_call(
   254→    base_url: str, api_key: str, model: str, messages: list[dict], timeout: float = 8.0,
   255→    max_tokens: int = 700
   256→) -> tuple[str, int]:
   257→    """Запрос к OpenAI-совместимому API. Возвращает (text, status_code).
   258→
   259→    max_tokens по умолчанию 700 — для диалоговых ответов это достаточно,
   260→    генерация быстрее, чем с 1024. Для раскладов вызывающий код явно
   261→    передаёт большее значение.
   262→    """
   263→    payload = {
   264→        "model": model,
   265→        "messages": messages,
   266→        "temperature": 0.85,
   267→        "max_tokens": max_tokens,
   268→        "top_p": 0.92,
   269→    }
   270→    headers = {
   271→        "Authorization": f"Bearer {api_key}",
   272→        "Content-Type": "application/json",
   273→    }
   274→    try:
   275→        async with httpx.AsyncClient(timeout=timeout) as client:
   276→            resp = await client.post(base_url, headers=headers, json=payload)
   277→            if resp.status_code == 429:
   278→                logger.warning(f"Rate limited on {base_url} ({model})")
   279→                return ("", 429)
   280→            if resp.status_code == 401:
   281→                logger.error(f"Auth failed for {base_url}")
   282→                return ("", 401)
   283→            if resp.status_code == 404:
   284→                logger.error(f"Model not found on {base_url} ({model})")
   285→                return ("", 404)
   286→            if resp.status_code == 400:
   287→                logger.error(f"Bad request on {base_url} ({model}): {resp.text[:200]}")
   288→                return ("", 400)
   289→            if resp.status_code == 503:
   290→                logger.warning(f"Service unavailable on {base_url} ({model})")
   291→                return ("", 503)
   292→            resp.raise_for_status()
   293→            data = resp.json()
   294→            if data.get("choices") and data["choices"][0].get("message"):
   295→                content = data["choices"][0]["message"].get("content", "")
   296→                if content:
   297→                    return (content.strip(), 200)
   298→            logger.warning(f"No text from {base_url}: {json.dumps(data)[:200]}")
   299→            return ("Извини, мне нужно подумать... Попробуй ещё раз.", 200)
   300→    except httpx.TimeoutException:
   301→        return ("", -1)
   302→    except Exception as e:
   303→        logger.error(f"OpenAI-compatible error ({model}): {e}")
   304→        return ("", -2)

   307→def _contents_to_messages(contents: list[dict]) -> list[dict]:
   308→    """Конвертирует Gemini contents format в OpenAI messages format."""
   309→    messages = []
   310→    for c in contents:
   311→        role = c.get("role", "user")
   312→        if role == "model":
   313→            role = "assistant"
   314→        parts = c.get("parts", [])
   315→        text = " ".join(p.get("text", "") for p in parts if p.get("text"))
   316→        if text:
   317→            messages.append({"role": role, "content": text})
   318→    return messages

   389→def _build_context(name: str, birth_date: str, facts: list[dict], history: list[dict],
   390→                   emotional: list[dict] = None, gender: str = "", age_group: str = "",
   391→                   zodiac: str = "", conversation_summary: str = "") -> str:
   392→    """Собираем контекст для LLM с эмоциональной памятью и зодиаком."""
   393→    lines = [f"Имя пользователя: {name}"]
   394→    if birth_date:
   395→        lines.append(f"Дата рождения: {birth_date}")
   396→    if zodiac:
   397→        lines.append(f"Знак зодиака: {zodiac}")
   398→    if gender:
   399→        lines.append(f"Пол (предположительно): {gender}")
   400→    if age_group:
   401→        age_map = {
   402→            "young": "до 18", "young_adult": "18-30", "adult": "30-45",
   403→            "mature": "45-60", "senior": "60+", "unknown": "неизвестно"
   404→        }
   405→        lines.append(f"Возрастная группа: {age_map.get(age_group, age_group)}")
   406→
   407→    if facts:
   408→        lines.append("Важные факты:")
   409→        for f in facts:
   410→            lines.append(f"- {f['fact_type']}: {f['fact_content']}")
   411→
   412→    if emotional:
   413→        type_names = {
   414→            "main_pain": "Главная боль", "loved_one": "Близкий человек",
   415→            "promise": "Обещание (себе)", "unfinished_question": "Незакрытый вопрос",
   416→            "life_event": "Событие жизни", "fear": "Страх", "goal": "Цель",
   417→            "breakthrough": "Прорыв",
   418→        }
   419→        lines.append("Эмоциональная память:")
   420→        for em in emotional[:5]:
   421→            label = type_names.get(em["memory_type"], em["memory_type"])
   422→            lines.append(f"- {label}: {em['content']}")
   423→
   424→    if conversation_summary:
   425→        lines.append(f"Сводка более раннего диалога: {conversation_summary}")
   426→
   427→    if history:
   428→        lines.append("Последние сообщения:")
   429→        for h in history[-6:]:
   430→            who = "София" if h["role"] == "sofia" else name
   431→            lines.append(f"{who}: {h['content'][:150]}")
   432→    return "\n".join(lines)

  1007→# ──────────────────────── Надёжный парсер JSON из LLM ────────────────────────
  1008→
  1009→def _parse_json_array(text: str, valid_field: str = "", valid_values: set = None,
  1010→                      max_items: int = 3, content_field: str = "") -> list[dict]:
  1011→    """Более надёжный парсер JSON-массива из ответа LLM.
  1012→
  1013→    Пытается несколькими способами:
  1014→    1. Прямой json.loads всего текста
  1015→    2. Извлечение по первому [ и последнему ]
  1016→    3. Извлечение по markdown-блоку ```json ... ```
  1017→    """
  1018→    import re
  1019→
  1020→    # Попытка 1: прямой парсинг
  1021→    try:
  1022→        data = json.loads(text.strip())
  1023→        if isinstance(data, list):
  1024→            return _filter_items(data, valid_field, valid_values, max_items, content_field)
  1025→    except (json.JSONDecodeError, ValueError):
  1026→        pass
  1027→
  1028→    # Попытка 2: markdown-блок ```json ... ```
  1029→    md_match = re.search(r'```(?:json)?\s*\n?(.*?)\n?```', text, re.DOTALL)
  1030→    if md_match:
  1031→        try:
  1032→            data = json.loads(md_match.group(1).strip())
  1033→            if isinstance(data, list):
  1034→                return _filter_items(data, valid_field, valid_values, max_items, content_field)
  1035→        except (json.JSONDecodeError, ValueError):
  1036→            pass
  1037→
  1038→    # Попытка 3: первый [ ... последний ]
  1039→    start = text.find("[")
  1040→    end = text.rfind("]")
  1041→    if start != -1 and end != -1 and end > start:
  1042→        try:
  1043→            data = json.loads(text[start:end + 1])
  1044→            if isinstance(data, list):
  1045→                return _filter_items(data, valid_field, valid_values, max_items, content_field)
  1046→        except (json.JSONDecodeError, ValueError):
  1047→            pass
  1048→
  1049→    return []
  1050→
  1051→
  1052→def _filter_items(items: list, valid_field: str, valid_values: set,
  1053→                  max_items: int, content_field: str) -> list[dict]:
  1054→    """Фильтрует извлечённые факты по валидности."""
  1055→    result = []
  1056→    for item in items[:max_items]:
  1057→        if not isinstance(item, dict):
  1058→            continue
  1059→        if valid_field and valid_values:
  1060→            if item.get(valid_field) not in valid_values:
  1061→                continue
  1062→        cf = content_field or "fact_content"
  1063→        if not item.get(cf) and not item.get("content"):
  1064→            continue
  1065→        result.append(item)
  1066→    return result
