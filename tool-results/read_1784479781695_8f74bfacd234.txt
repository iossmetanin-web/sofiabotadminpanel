    92→async def _typing_loop(chat_id: int, context: ContextTypes.DEFAULT_TYPE) -> None:
    93→    """Фоновый цикл: каждые 4 секунды обновляет индикатор «печатает...».
    94→
    95→    Telegram сбрасывает chat_action через ~5 секунд, поэтому нужно
    96→    постоянно его обновлять во время долгих LLM-вызовов.
    97→    Останавливается через asyncio.CancelledError.
    98→    """
    99→    try:
   100→        while True:
   101→            await asyncio.sleep(4)
   102→            try:
   103→                await context.bot.send_chat_action(chat_id=chat_id, action="typing")
   104→            except Exception:
   105→                pass
   106→    except asyncio.CancelledError:
   107→        return
   108→
   109→
   110→async def _with_typing(update: Update, context: ContextTypes.DEFAULT_TYPE, coro):
   111→    """Запускает coro, показывая индикатор «печатает...» всё время выполнения.
   112→
   113→    - Немедленно отправляет typing (до любых DB/LLM операций)
   114→    - В фоне обновляет каждые 4 секунды
   115→    - Гарантированно отменяет фоновую задачу при выходе
   116→    """
   117→    chat = update.effective_chat
   118→    if chat is None:
   119→        return await coro
   120→
   121→    # Немедленно — до любых DB-вызовов и rate limit проверок
   122→    try:
   123→        await context.bot.send_chat_action(chat_id=chat.id, action="typing")
   124→    except Exception:
   125→        pass
   126→
   127→    task = asyncio.create_task(_typing_loop(chat.id, context))
   128→    try:
   129→        return await coro
   130→    finally:
   131→        task.cancel()
   132→        try:
   133→            await task
   134→        except (asyncio.CancelledError, Exception):
   135→            pass

   138→async def _send_long_message(update: Update, text: str, max_length: int = 4096,
   139→                              reply_markup: InlineKeyboardMarkup = None) -> None:
   140→    """Отправляет длинное сообщение, разбивая на части."""
   141→    if len(text) <= max_length:
   142→        await update.effective_message.reply_text(text, reply_markup=reply_markup)
   143→        return
   144→
   145→    parts = []
   146→    current = ""
   147→    for paragraph in text.split("\n\n"):
   148→        if len(current) + len(paragraph) + 2 > max_length:
   149→            if current:
   150→                parts.append(current)
   151→            current = paragraph
   152→        else:
   153→            current = current + "\n\n" + paragraph if current else paragraph
   154→    if current:
   155→        parts.append(current)
   156→
   157→    # reply_markup только к последней части
   158→    for i, part in enumerate(parts):
   159→        markup = reply_markup if i == len(parts) - 1 else None
   160→        await update.effective_message.reply_text(part, reply_markup=markup)

   163→async def _safe_reply(update: Update, text: str, reply_markup: InlineKeyboardMarkup = None):
   164→    """Безопасная отправка с проверкой на callback vs message."""
   165→    try:
   166→        if update.callback_query:
   167→            await update.callback_query.message.reply_text(text, reply_markup=reply_markup)
   168→        else:
   169→            await update.effective_message.reply_text(text, reply_markup=reply_markup)
   170→    except Exception as e:
   171→        logger.error(f"_safe_reply error: {e}")

   264→async def cmd_start(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
   265→    """Обработчик команды /start. Поддержка рефералов: /start ref_<UID>."""
   266→    if not update.message or not update.effective_user:
   267→        return
   268→    await _with_typing(update, context, _cmd_start_inner(update, context))

   270→
   271→async def _cmd_start_inner(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
   272→    user_id = update.effective_user.id
   273→    username = update.effective_user.username
   274→    first_name = update.effective_user.first_name
   275→
   276→    user = await db.get_or_create_user(user_id, username, first_name)
   277→    await db.touch_last_seen(user_id)
   278→
   279→    # ─── Реферальная обработка ───
   280→    args = context.args if context.args else []
   281→    if args and args[0].startswith("ref_"):
   282→        try:
   283→            ref_uid = int(args[0][4:])
   284→            if user.get("message_count", 0) == 0 and ref_uid != user_id:
   285→                ref_user = await db.get_user(ref_uid)
   286→                if ref_user and not await db.was_referred(user_id):
   287→                    await db.set_referred_by(user_id, ref_uid)
   288→                    await db.add_crystals(
   289→                        ref_uid, 1,
   290→                        f"Реферальный бонус: пригласил {user_id}",
   291→                        txn_type="add",
   292→                    )
   293→                    await db.mark_referral_reward_given(user_id)
   294→                    logger.info(f"Referral: {user_id} referred by {ref_uid}, +1 💎 to {ref_uid}")
   295→        except (ValueError, TypeError):
   296→            pass
   297→
   298→    # Если пользователь уже был — перезапускаем с тёплым приветствием
   299→    if user.get("message_count", 0) > 0:
   300→        name = user.get("name") or first_name or "милый человек"
   301→        last_seen = user.get("last_seen_at")
   302→
   303→        if is_long_absence(last_seen, config.RETURN_ABSENCE_HOURS):
   304→            try:
   305→                facts = await db.get_memory_facts(user_id, min_importance=3)
   306→                emotional = await db.get_emotional_memory(user_id, min_importance=3)
   307→                last_topic = user.get("last_topic_summary") or ""
   308→                zodiac = _get_zodiac_from_user(user)
   309→                greeting = await generate_return_greeting(name, facts, emotional, last_topic, zodiac)
   310→                await update.effective_message.reply_text(greeting)
   311→                await db.save_message(user_id, "sofia", greeting, "return_greeting")
   312→            except Exception as e:
   313→                logger.error(f"Return greeting on /start error: {e}")
   314→                greeting = f"Здравствуй снова, {name}. Соскучилась по тебе. Что привело тебя ко мне сегодня?"
   315→                await update.effective_message.reply_text(greeting)
   316→                await db.save_message(user_id, "sofia", greeting, "greeting")
   317→        else:
   318→            greeting = (
   319→                f"Здравствуй снова, {name}. "
   320→                f"Соскучилась по тебе. Что привело тебя ко мне сегодня?"
   321→            )
   322→            await update.effective_message.reply_text(greeting)
   323→            await db.save_message(user_id, "sofia", greeting, "greeting")
   324→
   325→        await db.update_user_state(user_id, SofiaState.CONVERSATION)
   326→        await db.touch_last_seen(user_id)
   327→        return
   328→
   329→    # Первый вход — концепция v2: разрушение ожидания бота
   330→    greeting = (
   331→        "🌙\n\nЗдравствуй...\n\n"
   332→        "Я София.\n\n"
   333→        "Не знаю, что именно привело тебя сюда сегодня, но случайных встреч бывает меньше, чем нам кажется.\n\n"
   334→        "Как мне к тебе обращаться?"
   335→    )
   336→    await update.effective_message.reply_text(greeting)
   337→    await db.update_user_state(user_id, SofiaState.ASK_NAME)
   338→    await db.save_message(user_id, "sofia", greeting, "greeting")
   339→    logger.info(f"User {user_id} started the bot")

   434→        target_username = parts[2].lstrip("@")
   435→        amount = 0

   446→                target = s
   447→                break
   448→
   449→        if not target:
   450→            await update.effective_message.reply_text(f"Пользователь @{target_username} не найден.")
   451→            return
   452→
   453→        await db.add_crystals(
   454→            target["user_id"], amount,
   455→            f"Admin gift from {user_id}", txn_type="admin_gift",
   456→        )
   457→        new_balance = await db.get_user_crystals(target["user_id"])
   458→        await update.effective_message.reply_text(
   459→            f"✅ Начислено {amount} 💎 пользователю @{target_username}.\n"
   460→            f"Новый баланс: {new_balance} 💎"

   726→async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
   727→    """Главный обработчик всех текстовых сообщений."""
   728→    if not update.message or not update.effective_user:

   731→    await _with_typing(update, context, _handle_message_inner(update, context))

   734→async def _handle_message_inner(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
   735→    """Внутренняя логика handle_message, запускается под typing-индикатором."""
   736→    user_id = update.effective_user.id
   737→    text = update.message.text or ""
   738→    text_stripped = text.strip()
   739→    text_lower = text_stripped.lower()
   740→
   741→    # ─── Rate Limiting (ДО сохранения сообщения!) ───
   742→    if not await _check_rate_limit(user_id):
   743→        await update.effective_message.reply_text("Подожди чуток, милый человек... Я не успеваю.")
   744→        return
   745→
   746→    # ─── Получаем пользователя ───
   747→    user = await db.get_or_create_user(
   748→        user_id,
   749→        update.effective_user.username,
   750→        update.effective_user.first_name,
   751→    )
   752→    state = user.get("state", SofiaState.START)
   753→    is_blocked = user.get("is_blocked", False)
   754→
   755→    # ─── Сохраняем сообщение пользователя (только после rate limit) ───
   756→    await db.save_message(user_id, "user", text_stripped)
   757→
   758→    # ─── Проверка на «извини» (снимает блокировку) ───
   759→    if any(trigger in text_lower for trigger in SORRY_TRIGGERS):
   760→        if is_blocked or user.get("rudeness_count", 0) > 0:
   761→            await db.reset_rudeness(user_id)
   762→            await db.update_user_state(user_id, SofiaState.CONVERSATION)
   763→            response = "Ладно, милый человек. Всё забыто. Давай начнём сначала. О чём хочешь поговорить?"
   764→            await update.effective_message.reply_text(response)
   765→            await db.save_message(user_id, "sofia", response, "forgiveness")
   766→            return
   767→
   768→    # ─── Блокировка за грубость ───
   769→    if is_blocked:
   770→        await update.effective_message.reply_text(
   771→            "Я пока не готова продолжать разговор. Если хочешь поговорить спокойно — скажи «извини»."
   772→        )
   773→        return
   774→
   775→    # ─── Специальные текстовые команды (только в свободном диалоге/после онбординга) ───
   776→    if state in (SofiaState.CONVERSATION, SofiaState.PAID_HOOK, SofiaState.TARO_ASK_NUMBERS):
   777→        if any(trigger in text_lower for trigger in BALANCE_TRIGGERS):
   778→            await cmd_balance(update, context)
   779→            return
   780→        if any(trigger in text_lower for trigger in PROFILE_TRIGGERS):
   781→            await cmd_profile(update, context)
   782→            return
   783→        if any(trigger in text_lower for trigger in HISTORY_TRIGGERS):
   784→            await _show_history(update, user_id)
   785→            return
   786→        if any(trigger in text_lower for trigger in MENU_TRIGGERS):
   787→            await _show_menu(update, user)
   788→            return
   789→
   790→    # ─── Проверка на грубость ───
   791→    if is_rude(text_stripped):

   793→        idx = min(new_count - 1, len(RUDENESS_RESPONSES) - 1)

   795→
   796→        if new_count >= config.MAX_RUDENESS_BEFORE_BLOCK:
   797→            await db.update_user_profile(user_id, is_blocked=True)
   798→            await db.update_user_state(user_id, SofiaState.BLOCKED)
   799→

   801→        await db.save_message(user_id, "sofia", response, "rudeness")
   802→        return
   803→
   804→    # ─── Логика долгого отсутствия (только в CONVERSATION) ───
   805→    last_seen = user.get("last_seen_at")
   806→    if state == SofiaState.CONVERSATION and is_long_absence(last_seen, config.RETURN_ABSENCE_HOURS):
   807→        await _handle_return(update, user_id, user, text_stripped)
   808→        return
   809→
   810→    # ─── Бесплатная 1-карта (в диалоге) ───
   811→    if state == SofiaState.CONVERSATION and wants_free_card(text_lower):
   812→        await _handle_free_card(update, user_id, user)
   813→        return
   814→
   815→    # ─── Карта дня (Round 5, в диалоге) ───
   816→    if state == SofiaState.CONVERSATION and wants_card_of_day(text_lower):
   817→        await _handle_card_of_day(update, user_id, user)
   818→        return
   819→
   820→    # ─── Конкретный тип расклада по триггеру (love/career/decision/small/full/horoscope) ───
   821→    if state == SofiaState.CONVERSATION:
   822→        reading_type_state = detect_reading_type(text_lower)
   823→        if reading_type_state is not None:
   824→            # Переводим в TARO_ASK_NUMBERS — там есть полная логика выбора типа и запроса чисел
   825→            await db.update_user_state(user_id, SofiaState.TARO_ASK_NUMBERS)
   826→            await _handle_taro_numbers(update, user_id, text_stripped, user)
   827→            return
   828→

   830→    if state == SofiaState.CONVERSATION and wants_reading(text_lower):
   831→        crystals = user.get("crystals", 0)
   832→        name = user.get("name") or user.get("first_name") or "милый человек"
   833→        response = (
   834→            f"Хорошо, {name}. Карты сегодня особенно разговорчивы.\n\n"
   835→            f"Выбери, какой расклад тебе ближе — кнопкой ниже или напиши словами:\n\n"
   836→            f"❤️ Расклад на любовь — {config.TARO_LOVE_COST} 💎\n"
   837→            f"⚖️ Расклад на выбор — {config.TARO_DECISION_COST} 💎\n"
   838→            f"💼 Расклад на дело — {config.TARO_CAREER_COST} 💎\n"
   839→            f"🔮 Малый расклад (5 карт) — {config.TARO_SMALL_COST} 💎\n"

  1872→def _parse_date(text: str):
  1873→    from datetime import date as date_type
  1874→
  1875→    text = text.strip().replace(",", ".").replace("-", ".").replace("/", ".")
  1876→    formats = ["%d.%m.%Y", "%d.%m.%y", "%Y.%m.%d"]
  1877→
  1878→    for fmt in formats:
  1879→        try:
  1880→            parsed = datetime.strptime(text, fmt).date()
  1881→            current_year = datetime.now().year
  1882→            if 1900 <= parsed.year <= current_year:
  1883→                return parsed
  1884→        except ValueError:
  1885→            continue
  1886→
  1887→    match = re.search(r"(\d{1,2})[.\-/](\d{1,2})[.\-/](\d{2,4})", text)
  1888→    if match:
  1889→        try:
  1890→            day, month, year = int(match.group(1)), int(match.group(2)), int(match.group(3))
  1891→            if year < 100:
  1892→                year += 1900 if year >= 40 else 2000
  1893→            current_year = datetime.now().year
  1894→            if 1900 <= year <= current_year and 1 <= month <= 12 and 1 <= day <= 31:
  1895→                return date_type(year, month, day)
  1896→        except (ValueError, TypeError):
  1897→            pass
  1898→
  1899→    return None
  1900→

  1902→def _parse_time(text: str):
  1903→    from datetime import time as time_type
  1904→
  1905→    text = text.strip()
  1906→    for fmt in ["%H:%M", "%H:%M:%S", "%H"]:
  1907→        try:
  1908→            parsed = datetime.strptime(text, fmt)
  1909→            return parsed.time().replace(second=0, microsecond=0)
  1910→        except ValueError:
  1911→            continue
  1912→
  1913→    match = re.search(r"(\d{1,2})[:ч]\s*(\d{2})?", text)
  1914→    if match:
  1915→        try:
  1916→            hour = int(match.group(1))
  1917→            minute = int(match.group(2)) if match.group(2) else 0
  1918→            if 0 <= hour <= 23 and 0 <= minute <= 59:
  1919→                return time_type(hour, minute)
  1920→        except (ValueError, TypeError):
  1921→            pass
  1922→
  1923→    return None
